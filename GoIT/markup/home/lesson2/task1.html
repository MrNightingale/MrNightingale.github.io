// Function that shows value of factorial
function factorial(x) {
if (x >= 1) {
var result = x;
for(var i = 1; i < x; i++) {
result *= (x-i);
}
console.log(result);
}
else if (x == 0) {
console.log(1);
}
else {
throw new Error ('Value can not be negative or you should enter a number');
}
}
factorial(3);
